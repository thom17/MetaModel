@startuml
group UC01 Play Game
	Player->GameSystem : playGame(name)
	GameSystem->GameSystem : diceSetting()
	group main()
		group diceRoll(dice)
			GameSystem->Dice : roll()
		Dice-->GameSystem : Returns the result of the dice.
		end
		GameSystem->GameSystem : move(dice,player)
	GameSystem-->GameSystem : End of the game.
	end
	GameSystem->GameSystem : gameEnd()
end
group UC02 View Score
	Player->GameSystem : readRecord()
	GameSystem->GameSystem : printRecord()
end
@enduml
